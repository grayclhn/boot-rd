% Local sum, average, and scaled average macros.
% - 's' stands for 'set index'
% - 'o' stands for 'ordered index'

\newcommand\savg[3][\pm]{\tfrac{1}{M_{#1}({#3})} \ssum[#1]{#2}{#3}}
\newcommand\sclt[3][\pm]{\tfrac{1}{\sqrt{M_{#1}({#3})}} \ssum[#1]{#2}{#3}}
